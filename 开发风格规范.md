# CaseGo 前端开发风格规范

## 📋 规范目标

本规范旨在确保团队开发的代码具有以下特性：
- **代码简洁**：避免冗余代码，提高可读性
- **模块化设计**：逻辑清晰，职责分离
- **易于维护**：降低修复难度，提高开发效率
- **团队协作**：统一代码风格，减少沟通成本
- **性能优化**：避免重复请求，提升用户体验

### 2. Composables 业务逻辑封装
```javascript
// ✅ composables/useUserManagement.js
import { ref, reactive } from 'vue'
import { getUserList, createUser } from '@/api/user'
import { ElMessage } from 'element-plus'

export function useUserManagement() {
  const loading = ref(false)
  const userList = ref([])
  const queryParams = reactive({
    pageNum: 1,
    pageSize: 10,
    userName: ''
  })

  // 获取用户列表
  const fetchUserList = async () => {
    if (loading.value) return // 防止重复请求
    
    loading.value = true
    try {
      const response = await getUserList(queryParams)
      userList.value = response.rows
    } catch (error) {
      ElMessage.error('获取用户列表失败')
    } finally {
      loading.value = false
    }
  }

  // 创建用户
  const handleCreateUser = async (userData) => {
    try {
      await createUser(userData)
      ElMessage.success('创建成功')
      await fetchUserList() // 刷新列表
    } catch (error) {
      ElMessage.error('创建失败')
    }
  }

  return {
    loading,
    userList,
    queryParams,
    fetchUserList,
    handleCreateUser
  }
}
```

## 🚫 防重复请求机制

### 1. 在 Composables 中实现
```javascript
// ✅ 推荐：使用 loading 状态防止重复请求
export function useApiRequest() {
  const loading = ref(false)
  const requestCache = new Map()

  const makeRequest = async (apiFunction, params, cacheKey) => {
    // 防止重复请求
    if (loading.value) {
      console.warn('请求正在进行中，请勿重复提交')
      return
    }

    // 缓存机制（可选）
    if (cacheKey && requestCache.has(cacheKey)) {
      return requestCache.get(cacheKey)
    }

    loading.value = true
    try {
      const result = await apiFunction(params)
      if (cacheKey) {
        requestCache.set(cacheKey, result)
      }
      return result
    } finally {
      loading.value = false
    }
  }

  return { loading, makeRequest }
}
```

  <el-form 
    ref="formRef"
    :model="formData"
    :rules="formRules"
    label-width="100px"
  >
    <el-form-item label="用户名" prop="username">

  username: '',
  email: ''

  try {
    await formRef.value.validate()
    // 提交表单

    ElMessage.error('请检查表单输入')

## 📊 错误处理规范

### 1. 统一错误处理
```javascript
// ✅ composables/useErrorHandler.js
import { ElMessage, ElNotification } from 'element-plus'

export function useErrorHandler() {
  const handleApiError = (error, customMessage) => {
    console.error('API Error:', error)
    
    const message = customMessage || error.message || '操作失败'
    
    if (error.response?.status === 401) {
      ElMessage.error('登录已过期，请重新登录')
      // 跳转到登录页
    } else if (error.response?.status >= 500) {
      ElNotification.error({
        title: '服务器错误',
        message: '服务器暂时无法响应，请稍后重试'
      })
    } else {
      ElMessage.error(message)
    }
  }

  const handleValidationError = (errors) => {
    const firstError = Object.values(errors)[0]
    ElMessage.error(firstError[0])
  }

  return {
    handleApiError,
    handleValidationError
  }
}
```
